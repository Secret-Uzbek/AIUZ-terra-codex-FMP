import hashlib
import json
from datetime import datetime

class TerraCodexValidator:
    def __init__(self, codex_ontology="codex_terra.yaml"):
        self.ontology = self.load_ontology(codex_ontology)
        self.required_attributes = [
            'document_id', 'hash_signature', 'author_did',
            'creation_date', 'version', 'layer_compliance'
        ]
    
    def validate_document(self, document):
        """Полная валидация документа Terra Codex"""
        results = {
            'l0_validation': self.validate_l0_base(document),
            'l1_validation': self.validate_l1_semantic(document),
            'l2_validation': self.validate_l2_infrastructure(document),
            'l3_validation': self.validate_l3_governance(document),
            'l4_validation': self.validate_l4_interface(document)
        }
        return results
    
    def validate_l0_base(self, document):
        """L0: Базовая структура и идентификация"""
        checks = {
            'has_uuid': self.check_uuid_format(document.document_id),
            'hash_valid': self.verify_hash_signature(document),
            'qr_valid': self.verify_qr_code(document),
            'attributes_complete': self.check_required_attributes(document)
        }
        return all(checks.values()), checks
    
    def validate_l1_semantic(self, document):
        """L1: Семантическая согласованность"""
        checks = {
            'ontology_mapping': self.validate_ontology_refs(document),
            'thesaurus_terms': self.validate_thesaurus_refs(document),
            'semantic_tags': self.validate_semantic_tags(document),
            'content_coherence': self.check_content_coherence(document)
        }
        return all(checks.values()), checks
    
    def validate_l2_infrastructure(self, document):
        """L2: Инфраструктурная совместимость"""
        checks = {
            'storage_format': self.validate_storage_format(document),
            'terra_point_refs': self.validate_terra_point_refs(document),
            'file_structure': self.validate_file_structure(document),
            'encoding_utf8': self.validate_encoding(document)
        }
        return all(checks.values()), checks
    
    def validate_l3_governance(self, document):
        """L3: Управление и доступ"""
        checks = {
            'did_valid': self.validate_did_format(document.author_did),
            'access_roles': self.validate_access_roles(document),
            'dao_compliance': self.validate_dao_requirements(document),
            'version_control': self.validate_version_format(document)
        }
        return all(checks.values()), checks
    
    def validate_l4_interface(self, document):
        """L4: Интерфейсы и экспорт"""
        checks = {
            'api_compatibility': self.validate_api_endpoints(document),
            'export_formats': self.validate_export_formats(document),
            'multilingual': self.validate_multilingual_support(document),
            'user_interface': self.validate_ui_compliance(document)
        }
        return all(checks.values()), checks
    
    def generate_hash_signature(self, document):
        """Генерация SHA256 подписи"""
        content = f"{document.title}{document.author_did}{document.creation_date}{document.version}"
        hash_obj = hashlib.sha256(content.encode('utf-8'))
        return f"AIUZ-{hash_obj.hexdigest()}"
    
    def generate_qr_code(self, hash_signature):
        """Генерация QR кода для верификации"""
        return f"AIUZ://auth/{hash_signature.split('-')[1]}@aiuz2025.local"

class TerraCodexDocument:
    def __init__(self, module_name, author_did, lang="ru"):
        self.module_name = module_name
        self.author_did = author_did
        self.lang = lang
        self.creation_date = datetime.now().strftime("%Y-%m-%d")
        self.version = "1.0"
        self.document_id = self.generate_document_id()
        
    def generate_document_id(self):
        """Генерация уникального UUID документа"""
        module_number = self.get_next_module_number(self.module_name)
        return f"urn:terra-codex:{self.module_name}-{module_number:05d}"
    
    def create_metadata_header(self):
        """Создание заголовка с метаданными"""
        hash_sig = self.generate_hash_signature()
        qr_code = self.generate_qr_code(hash_sig)
        
        return f"""---
file: {self.module_name}.{self.lang}.v{self.version}.md
document_id: {self.document_id}
creation_date: {self.creation_date}
version: {self.version}
author_did: {self.author_did}
hash_signature: {hash_sig}
qr_verification: {qr_code}
conforms_to: TERRA_CODEX_Standard_v1.0
layer_compliance: [L0, L1, L2, L3, L4]
---"""
    
    def validate_before_publish(self):
        """Валидация перед публикацией"""
        validator = TerraCodexValidator()
        return validator.validate_document(self)
    
    def submit_to_dao(self):
        """Отправка на рассмотрение DAO"""
        if self.validate_before_publish():
            return self.dao_submission_process()
        else:
            raise ValidationError("Document failed validation")

API Спецификация Terra Codex

class TerraCodexAPI:
    """
    RESTful API для управления документацией Terra Codex
    """
    
    def __init__(self, base_url="https://api.terra-codex.org/v1"):
        self.base_url = base_url
        self.endpoints = {
            'documents': '/documents',
            'validate': '/validate',
            'parse': '/parse',
            'translate': '/translate',
            'dao': '/dao',
            'ontology': '/ontology'
        }
    
    def create_document(self, document_data):
        """POST /documents - Создание нового документа"""
        endpoint = f"{self.base_url}{self.endpoints['documents']}"
        return self.post_request(endpoint, document_data)
    
    def validate_document(self, document_id):
        """POST /validate - Валидация документа"""
        endpoint = f"{self.base_url}{self.endpoints['validate']}"
        return self.post_request(endpoint, {'document_id': document_id})
    
    def parse_semantic(self, content):
        """POST /parse - Семантический анализ"""
        endpoint = f"{self.base_url}{self.endpoints['parse']}"
        return self.post_request(endpoint, {'content': content})
    
    def translate_document(self, document_id, target_lang):
        """POST /translate - Перевод документа"""
        endpoint = f"{self.base_url}{self.endpoints['translate']}"
        return self.post_request(endpoint, {
            'document_id': document_id,
            'target_language': target_lang
        })
    
    def submit_to_dao(self, document_id, submission_type="review"):
        """POST /dao - Отправка в DAO"""
        endpoint = f"{self.base_url}{self.endpoints['dao']}"
        return self.post_request(endpoint, {
            'document_id': document_id,
            'type': submission_type
        })
    
    def query_ontology(self, term):
        """GET /ontology - Запрос к онтологии"""
        endpoint = f"{self.base_url}{self.endpoints['ontology']}"
        return self.get_request(f"{endpoint}?term={term}")

Интеграция с Terra Points

class TerraPointIntegration:
    """
    Интеграция документации с узлами Terra Points
    """
    
    def __init__(self):
        self.terra_points = self.discover_terra_points()
        self.replication_factor = 3
        
    def store_document(self, document, terra_points=None):
        """Распределённое хранение документа"""
        if not terra_points:
            terra_points = self.select_optimal_points(self.replication_factor)
        
        storage_results = []
        for point in terra_points:
            result = point.store_document(document)
            storage_results.append(result)
        
        return self.verify_storage_integrity(storage_results)
    
    def synchronize_document(self, document_id):
        """Синхронизация документа между узлами"""
        document_versions = {}
        for point in self.terra_points:
            version = point.get_document_version(document_id)
            document_versions[point.id] = version
        
        latest_version = self.determine_latest_version(document_versions)
        return self.sync_to_latest(document_id, latest_version)
    
    def validate_network_integrity(self):
        """Проверка целостности сети документов"""
        integrity_results = {}
        for point in self.terra_points:
            integrity_results[point.id] = point.validate_local_storage()
        
        return self.analyze_network_health(integrity_results)

class TerraCodexTestSuite:
    """
    Автоматизированные тесты для документации Terra Codex
    """
    
    def test_document_validation(self, document):
        """Тест валидации документа"""
        validator = TerraCodexValidator()
        result = validator.validate_document(document)
        assert result['l0_validation'][0], "L0 validation failed"
        assert result['l1_validation'][0], "L1 validation failed"
        assert result['l2_validation'][0], "L2 validation failed"
        assert result['l3_validation'][0], "L3 validation failed"
        assert result['l4_validation'][0], "L4 validation failed"
    
    def test_api_endpoints(self):
        """Тест API эндпоинтов"""
        api = TerraCodexAPI()
        endpoints = ['/documents', '/validate', '/parse', '/translate']
        for endpoint in endpoints:
            response = api.health_check(endpoint)
            assert response.status_code == 200
    
    def test_multilingual_consistency(self, document_id):
        """Тест мультиязычной согласованности"""
        languages = ['ru', 'uz-latin', 'en', 'de']
        consistency_scores = []
        
        for lang in languages:
            if self.document_exists(document_id, lang):
                score = self.check_semantic_consistency(document_id, lang)
                consistency_scores.append(score)
        
        avg_consistency = sum(consistency_scores) / len(consistency_scores)
        assert avg_consistency > 0.95, "Multilingual consistency below threshold"
    
    def test_terra_point_integration(self):
        """Тест интеграции с Terra Points"""
        integration = TerraPointIntegration()
        health_results = integration.validate_network_integrity()
        
        healthy_points = sum(1 for result in health_results.values() if result)
        total_points = len(health_results)
        health_ratio = healthy_points / total_points
        
        assert health_ratio > 0.9, "Terra Point network health below threshold"
