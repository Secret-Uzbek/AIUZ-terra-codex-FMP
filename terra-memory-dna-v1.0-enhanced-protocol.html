class DNAStrand:
    """–û—Ç–¥–µ–ª—å–Ω–∞—è –Ω–∏—Ç—å –î–ù–ö –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–≥–æ —Ç–∏–ø–∞ –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏"""
    
    def __init__(self, name: str, priority: str = "medium"):
        self.name = name
        self.priority = priority
        self.sequences = []
        self.metadata = {
            "creation_date": datetime.now().isoformat(),
            "last_accessed": datetime.now().isoformat(),
            "access_count": 0,
            "total_size": 0
        }
        
    def store_sequence(self, sequence_data: Dict[str, Any]):
        """–°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –≤ –Ω–∏—Ç–∏"""
        
        sequence = {
            "id": str(uuid.uuid4()),
            "timestamp": datetime.now().isoformat(),
            "data": sequence_data,
            "size": len(str(sequence_data)),
            "checksum": self._calculate_checksum(sequence_data)
        }
        
        self.sequences.append(sequence)
        self.metadata["total_size"] += sequence["size"]
        self.metadata["last_accessed"] = datetime.now().isoformat()
        
    def query_sequences(self, query_params: Dict[str, Any] = None) -> List[Dict[str, Any]]:
        """–ó–∞–ø—Ä–æ—Å –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π –ø–æ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º"""
        
        if not query_params:
            return self.sequences
            
        filtered_sequences = []
        
        for sequence in self.sequences:
            if self._matches_query(sequence, query_params):
                filtered_sequences.append(sequence)
                
        self.metadata["access_count"] += 1
        self.metadata["last_accessed"] = datetime.now().isoformat()
        
        return filtered_sequences
        
    def _matches_query(self, sequence: Dict[str, Any], 
                      query_params: Dict[str, Any]) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –∑–∞–ø—Ä–æ—Å—É"""
        
        for key, value in query_params.items():
            if key == "timestamp_range":
                seq_time = datetime.fromisoformat(sequence["timestamp"])
                start_time = datetime.fromisoformat(value["start"])
                end_time = datetime.fromisoformat(value["end"])
                
                if not (start_time <= seq_time <= end_time):
                    return False
                    
            elif key == "content_search":
                if value.lower() not in str(sequence["data"]).lower():
                    return False
                    
            elif key == "min_size":
                if sequence["size"] < value:
                    return False
                    
            elif key == "max_size":
                if sequence["size"] > value:
                    return False
                    
        return True
        
    def optimize_storage(self):
        """–û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤ –Ω–∏—Ç–∏"""
        
        # –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ –≤–∞–∂–Ω–æ—Å—Ç–∏ (–Ω–æ–≤—ã–µ - –≤–∞–∂–Ω–µ–µ)
        self.sequences.sort(key=lambda x: x["timestamp"], reverse=True)
        
        # –£–¥–∞–ª–µ–Ω–∏–µ –¥—É–±–ª–∏–∫–∞—Ç–æ–≤
        unique_sequences = []
        seen_checksums = set()
        
        for sequence in self.sequences:
            if sequence["checksum"] not in seen_checksums:
                unique_sequences.append(sequence)
                seen_checksums.add(sequence["checksum"])
                
        self.sequences = unique_sequences
        
        # –ü–µ—Ä–µ—Å—á–µ—Ç —Ä–∞–∑–º–µ—Ä–∞
        self.metadata["total_size"] = sum(seq["size"] for seq in self.sequences)
        
    def get_size(self) -> int:
        """–ü–æ–ª—É—á–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞ –Ω–∏—Ç–∏"""
        return self.metadata["total_size"]
        
    def get_sequence_count(self) -> int:
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π"""
        return len(self.sequences)
        
    def _calculate_checksum(self, data: Any) -> str:
        """–†–∞—Å—á–µ—Ç –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–æ–π —Å—É–º–º—ã –¥–∞–Ω–Ω—ã—Ö"""
        import hashlib
        return hashlib.md5(str(data).encode()).hexdigest()
